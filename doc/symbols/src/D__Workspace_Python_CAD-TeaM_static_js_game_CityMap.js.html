1 /**
  2  * @file Represents a city map object.
  3  * @author Cristian Sima
  4  * @version 30.12.2014
  5  */
  6 
  7 /**
  8  * It represents a cityMap object. It extends the XantenMap object
  9  * @constructor
 10  * @extends XantenMap
 11  * @param {array} obj The array of the map
 12  */
 13 function CityMap(obj) {
 14 		
 15 	var cityMap 						= new XantenMap(obj.array, 'city');
 16 	cityMap.backgrounds					= obj.backgrounds;
 17 	
 18 	/**
 19 	 * It is called when a cell is selected. 
 20 	 * @name selectCell
 21 	 * @memberOf CityMap.prototype
 22 	 * @param (number) x The x coordinate
 23 	 * @param (number) y The y coordinate
 24 	 */
 25 	cityMap.selectCell = function(x, y){
 26 
 27 		if($(this.HTML_element + " #cel_"+x+"_"+y).hasClass('allow_construction')){
 28 			game.performAction('available_buildings');
 29 			this.selectBackground(x, y);
 30 		}
 31 		else if($(this.HTML_element+" #cel_"+x+"_"+y+" img").hasClass('hasAction')){
 32 			var building = $(this.HTML_element+" #cel_"+x+"_"+y+" img").attr("name_of_building");
 33 			game.performAction('inside_building', {name: building});
 34 			this.selectImage(x, y);
 35 		}else {
 36 			game.performAction('city-map-selected', null);
 37 		}
 38 	};
 39 		
 40 	/**
 41 	 * It renders the information of the array in the images
 42 	 * @name render
 43 	 * @memberOf CityMap.prototype
 44 	 */
 45 	cityMap.render = function(){		
 46 		for(var i=1; i <= this.array.length; i++){
 47 			for(var j=1; j <= this.array.length; j++){
 48 				var cell 		= $(this.HTML_element + " #cel_"+i+"_"+j),
 49 					background 	= this.backgrounds[this.array[(i-1)][(j-1)].id_background];				
 50 				
 51 				cell.removeClass();				
 52 				
 53 				if(background.allowBuildings === true) {					
 54 					if(background.allowBuildings === true && this.array[(i-1)][(j-1)].type_construction === null){			
 55 						cell.addClass("allow_construction");	
 56 					}
 57 					else if(this.array[(i-1)][(j-1)].type_construction === "building") {
 58 						var building 		= game.constructions.getBuildingById(this.array[(i-1)][(j-1)].id_construction),
 59 							img		 		= building.name,
 60 							player_building = game.player.city.buildings[building.name];
 61 						
 62 						// free commercial image from https://openclipart.org/detail/149155/old-mine-by-chrisdesign
 63 						if(player_building.status === 'under_construction'){
 64 							img = "construction";
 65 						}
 66 						cell.html(HTML_Engine.getBuilding.image(img, game.player.getLevel(), 98) + (player_building.level?"<span class='" + ((player_building.status === 'upgrading')?"level_upgrading":"level") + "' >" + player_building.level + "</span>":""));
 67 						
 68 						$(this.HTML_element+" #cel_"+i+"_"+j+" img").addClass('hasAction');	
 69 						$(this.HTML_element+" #cel_"+i+"_"+j+" img").attr({"name_of_building": building.name.toLowerCase()});	
 70 						$(this.HTML_element+" #cel_"+i+"_"+j+" img").attr({"title": building.name.capitalize()});	
 71 					} else {
 72 						var element = game.constructions.getElementById(this.array[(i-1)][(j-1)].id_construction);
 73 						cell.html(HTML_Engine.getBuilding.image(element.name + "", game.player.getLevel(), 70));
 74 						$(this.HTML_element+" #cel_"+i+"_"+j+" img").attr({"name_of_building": element.name.toLowerCase()});	
 75 						$(this.HTML_element+" #cel_"+i+"_"+j+" img").attr({"title": element.name.capitalize()});
 76 					}					
 77 				} 			
 78 				$(this.HTML_element + " #cel_"+i+"_"+j).css('background-image', 'url(/static/img/game/background/city/' + background.img + ')');
 79 			}
 80 		}
 81 	};	
 82 	return cityMap;
 83 }
