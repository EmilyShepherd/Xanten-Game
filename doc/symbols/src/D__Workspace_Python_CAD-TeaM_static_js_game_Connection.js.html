 86 Connection.prototype.startGame = function(response) {
 87 
 88 
 89 
 90 	clearTimeout(this.threadUsers);

 92 	
 93 	var path 		= "/static/js/game/",
 94 		/**
 95 		 * It loads a JavaScript file
 96 		 */
 97 		loadJsFile 	= function(file) {
 98 			var jsLink = $("<script type='text/javascript' src='" + file + "'>");
 99 			$("head").append(jsLink); 
100 		},
101 		/**
102 		 * It loads the JavaScript objects for game
103 		 */
104 		loadJS 		= function() {
105 			var files 		= [

117 				          		   path + "Task.js",

132 				player		= new Player(response.player);
133 		
134 			/*
135 			 * It creates the game
136 			 */
137 			game 		= new Game(response.game.data, player, cityMap, worldMap, response.game.tasks);
138 	
139 			/*
140 			 * Loads the libraries
141 			 */
142 			var files = [
143 			             path + "game.constructions.js",
144 			             path + "game.organization.js",
145 			             path + "game.unit.js",
146 			             path + "game.actions.js",
147 			             path + "game.tasks.js"],

164 	inc.href   = '/static/css/gamestyle.css';
165 	inc.onload = loadJS;
166 
167 	document.head.appendChild(inc);

270 /**
271  * It helps a user to join a game based on the token. If there are problems it displays a message
272  * @param {string} token The token of the game
273  * @memberOf Connection.prototype
274  */
275 Connection.prototype.joinGameNow = function(token) {	
276 	if (!token) {
277 		return;
278 	}
279 	var name = prompt("Choose your name in the game...", "");
280 	name = (name)?name:"";
281 	clearInterval(this.threadGames);
282 	request = $.ajax({
283 		url: "/game/"+token+"/join",
284 		type: "PUT",
285 		data: {"username":name},
286 		success: function (response) {
287 			if(response['status'] === "error") {
288 				alert(response['message']);
289 				connection.threadGames = setInterval(connection.getGames, 2000);
290 			}
291 			else {
292 				$("#game_start").hide();
293 				$("#map_div").hide();
294 				$("#join_game").hide();
295 				$("#start_game").hide();
296 				$("#create_game").show(500, function() {
297 					connection.waitingPlayers(response);					
298 				})
299 			}
300 		}
301 	})
302 };

342 	/**
343 	 *  Checks the value of a space in the map array, returning the empty string for co-ordinates that don't exist.
344 	 * (This avoids nasty array index errors when we get close to the edges.)
345 	 * @param {string} y_coord The y co-ordonate
346 	 * @param {string}x_coord The x co-ordonate

348 	 * @author Joe
349 	 */
350 	check = function(y_coord, x_coord) {
351 		if (x_coord < 0 || y_coord < 0 || x_coord > 6 || y_coord > 6) {
352 			return "";
353 		} else {
354 			return imgArray[y_coord][x_coord];
355 		}
356 	}
357 
358 
359 	// Controlling amount of space on the map.
360 	var minClear = 0,
361 		noClear = 49;
362 	
363 	switch (size) {
364 		// Small: 2-4 players
365 		// Medium: 4-6 players
366 		// Large: 6-10 players
367 		case "small":
368 			minClear = 8;
369 			break;
370 		case "large":
371 			minClear = 20;
372 			break;
373 		default:
374 			minClear = 12;
375 		}
376 
377 	// Variables which control properties of the world being generated:
378 	// How often do things appear?
379 	var OCEAN_SEED = Math.floor((Math.random() * 9) + 1)/100, // 0.01 ---> 0.09
380 	 	ROCK_SEED = 0.01,
381 	 	FOREST_SEED = 0.1,
382 	// What's the likelihood that they'll spread?
383 	 	OCEAN_EXPAND = Math.floor((Math.random() * 9) + 3)/100, // 0.03 --> 0.09
384 	 	FOREST_EXPAND = Math.floor((Math.random() * 9) + 5)/100, // 0.05 --> 0.09
385 	// Calculating probabilities:
386 	 	p_ocean = 1 - OCEAN_SEED,
387 	 	p_rock = p_ocean - ROCK_SEED,
388 	 	p_forest = p_rock - FOREST_SEED,
389 
390 		 x = 0,

393 	
394 	while (y < 7) {
395 		while (x < 7) {
396 			r = Math.random();
397 			if ((r > p_ocean) && (noClear > minClear)) {
398 				imgArray[y][x] = "o";
399 				noClear--;
400 			} else {
401 				imgArray[y][x] = "g";
402 			}
403 			x++;
404 		}

406 		y++;
407 	}

411 		while (x < 7) {

414 					if (x > 0) {
415 						if (Math.random() < OCEAN_EXPAND) {
416 							imgArray[y][x - 1] = "o";
417 							noClear--;
418 						}
419 					}
420 					if (x < 6) {
421 							if (Math.random() < OCEAN_EXPAND) {
422 							imgArray[y][x + 1] = "o";

424 						}
425 					}

427 						if (Math.random() < OCEAN_EXPAND) {
428 							imgArray[y - 1][x] = "o";

430 						}

444 	}
445 	x = 0;

450 				if (check(y+1, x) === "o") {
451 					if (check(y-1, x) === "o") {

460 					} else {

606 			}

615 				}
616 			}
617 			x++;
618 		}
619 		x = 0;
620 		y++;
621 	}
622 	// Straight borders
623 	x = 0;
624 	y = 0;

650 					} else {
651 						imgArray[y][x] = "o-s";
652 					}

659 						imgArray[y][x] = "o-e";

674 				if (check(y, x-1) === "o-s") imgArray[y][x] = "oa-ne";
675 				if (check(y+1, x) === "o-w") imgArray[y][x] = "oa-ne";
676 			}
677 			if (check(y, x) === "oc-nw") {
678 				if (check(y, x+1) === "o-s") imgArray[y][x] = "oa-nw";
679 				if (check(y+1, x) === "o-e") imgArray[y][x] = "oa-nw";

682 				if (check(y, x-1) === "o-n") imgArray[y][x] = "oa-ne";

688 			}
689 			if (check(y, x) === "o") {
690 				if (check(y-1, x) === "ob-nw" && check(y, x+1) === "oc-sw") imgArray[y][x] = "ob-sw";
691 				if (check(y+1, x) === "ob-sw" && check(y, x+1) === "oc-nw") imgArray[y][x] = "ob-nw";
692 				if (check(y-1, x) === "ob-ne" && check(y, x-1) === "oc-se") imgArray[y][x] = "ob-se";
693 				if (check(y+1, x) === "ob-se" && check(y, x-1) === "oc-ne") imgArray[y][x] = "ob-ne";
694 			}
695 			if (check(y, x) === "g") {
696 				r = Math.random();
697 				if (r < FOREST_SEED/2) {
698 					imgArray[y][x] = "t0";
699 				} else if (r < FOREST_SEED) {
700 					imgArray[y][x] = "t1";
701 				}
702 			}
703 			x++;
704 		}
705 		x = 0;
706 		y++;
707 	}
708 	x = 0;
709 	y = 0;
710 	while (y < 7) {
711 		while (x < 7) {
712 			if (check(y, x) === "t0") {
713 				if (check(y-1,x) === "g" && Math.random() < FOREST_EXPAND) imgArray[y-1][x] = "t1";
714 				if (check(y+1,x) === "g" && Math.random() < FOREST_EXPAND) imgArray[y+1][x] = "t1";
715 				if (check(y,x-1) === "g" && Math.random() < FOREST_EXPAND) imgArray[y][x-1] = "t1";
716 				if (check(y,x+1) === "g" && Math.random() < FOREST_EXPAND) imgArray[y][x+1] = "t1";
717 			}
718 			if (check(y, x) === "g" && Math.random() < ROCK_SEED) imgArray[y][x] = "m".concat(Math.round(Math.random()));
719 			x++;
720 		}
721 		x = 0;
722 		y++;
723 	}
724 	return imgArray;
725 };
