 20 /**
 21  * It returns a short representation of the resource.
 22  * @example
 23  * <1000 => the value
 24  * 1k
 25  * 1m
 26  * 1b
 27  * @memberOf HTML_Engine
 28  * @param Number input The amount of resource
 29  * @returns string The amount in a short representation
 30  */
 31 HTML_Engine.shortResourceRepresentation = function(input) {
 32 	var len = parseInt(input.toString().length),
 33 		text = "";
 34 
 35 	if (len <= 3) {
 36 		text = input;
 37 	} else
 38 	if (len >= 4 && len <= 6) {
 39 		text = ((input / 1000 * 100) / 100).toFixed(2) + "k";
 40 	} else
 41 	if (len >= 7 && len <= 9) {
 42 		text = ((input / 1000000 * 100) / 100).toFixed(2) + 'm';
 43 	} else
 44 		text = ((input / 1000000000 * 100) / 100).toFixed(2) + 'b';
 45 
 46 	return "<span title='" + input.formatNumber(0) + "'>" + text + "</span>";
 47 };

 49 /**
 50  * It returns a short representation of time in this format: {hours} h, {minutes} h, {seconds} sec OR "Instant" (if time is 0)
 51  * @memberOf HTML_Engine
 52  * @param Number sec_num The time expressed in number of seconds
 53  * @return string The time in this format: {hours} h, {minutes} h, {seconds} sec OR "Instant" (if time is 0)
 54  */
 55 HTML_Engine.shortTimeRepresentation = function(sec_num) {
 56 	sec_num = parseInt(sec_num);
 57 
 58 	if (sec_num === 0) {
 59 		return "<span class='bold'>Instant</span>";
 60 	}
 61 	var elements = [],
 62 		hours = Math.floor(sec_num / 3600),
 63 		minutes = Math.floor((sec_num - (hours * 3600)) / 60),
 64 		seconds = sec_num - (hours * 3600) - (minutes * 60);
 65 
 66 	if (hours !== 0) {
 67 		elements.push("<span class='bold'>" + hours + " h</span>");
 68 	}
 69 	if (minutes !== 0) {
 70 		elements.push("<span class='bold'>" + minutes + " min</span>");
 71 	}
 72 	if (seconds !== 0) {
 73 		elements.push("<span class='bold'>" + seconds + " sec</span>");
 74 	}
 75 
 76 	return elements.join(", ");
 77 };

229 /**
230  * @namespace A chooser represents a set of mutiple input values from user using sliders
231  * @memberOf HTML_Engine
232  */
233 HTML_Engine.chooser = {
234 	/**
235 	 * It generates the html of the chooser
236 	 * @param {object} args Information regarding the chooser. It must contain:
237 	 * @example The id of the chooser: chooser.id
238 	 * A description for the chooser: chooser.info
239 	 * The elements of the choose as an array, each object with an id and title: chooser.values
240 	 * The title of the button: chooser.button
241 	 * @return {string} The HTML code for the chooser
242 	 */
243 	content: function(args) {
244 		var html = '';
245 		html = "<div class='heading'>" +
246 			"<div class='chooser' id='" + args.id + "_chooser' > " +
247 			"<div class='info'>" + (args.info ? args.info : "") + "</div>";
248 		for (var i = 0; i < args.values.length; i++) {
249 			var value = args.values[i];
250 			html += "<div class='element' id='element_" + value.id + "'>" +
251 				"<div class='bold'>" + value.title + ":</div>" +
252 				"<div id='extra'></div>" +
253 				"<div><div id='slider' class='element_slider'></div></div>" +
254 				"<div>" +
255 				"<input type='number' size='3' value=''  id='input_value' />" +
256 				"</div>" +
257 				"</div>";
258 		}
259 		html += "<div class='center'>" +
260 			"<input type='button' value='" + args.button + "' id='" + args.id + "_button' /> " +
261 			"</div>" +
262 			"</div>" +
263 			"</div>";
264 		return html;
265 	},
266 	/**
267 	 * It enables the chooser. It enables all the elements inside it
268 	 * @param {object} args An object with the settings for all the elements inside a chooser:
269 	 * @example The id of the chooser: args.id
270 	 * The values of the elements as an array: args.values. Each element contain: id, min, max
271 	 * The action to be performed when the button is clicked: chooser.performAction
272 	 */
273 	enable: function(args) {
274 
275 		$.each(args.values, function(index, value) {
276 
277 			value = args.values[index];
278 
279 			var id = args.id,
280 				div = $("#" + id + "_chooser #element_" + value.id),
281 				input = div.find("#input_value"),
282 				slider = div.find("#slider"),
283 				extra = div.find("#extra"),
284 				action = value.change ? value.change : undefined,
285 				gen_act = args.generalChange ? args.generalChange : undefined,
286 				change = function(event, ui) {
287 					input.val(ui.value);
288 					if (action) {
289 						action(event, ui, extra);
290 					}
291 					if (gen_act) {
292 						gen_act(event, ui, extra);
293 					}
294 				};
295 
296 			slider.slider({
297 				min: value.min ? value.min : 0,
298 				max: value.max ? value.max : 0,
299 				animate: true,
300 				range: "min",
301 				value: value.min ? value.min : 0,
302 				slide: change,
303 				change: change,
304 				create: function(event, ui) {
305 					ui.value = $(this).slider("option", "min");
306 					change(event, ui);
307 				}
308 			});
309 
310 			input.on("change keyup paste", function() {
311 				if ($(this).val() !== '' && (parseInt($(this).val()) || $(this).val() === '0')) {
312 					slider.slider("value", parseInt($(this).val()));
313 				}
314 			});
315 
316 
317 		});
318 
319 		$("#" + args.id + "_button")
320 			.button()
321 			.click(args.performAction);
322 	},
323 	/**
324 	 * It removes the functions from the memory and disables the listenerrs
325 	 * @param {object} args An object with the id of the chooser.
326 	 */
327 	disable: function(args) {
328 
329 		var divs = $("#" + args.id + "_chooser div.element");
330 
331 		for (var i = 0; i < divs.length; i++) {
332 			var div = divs[i];
333 			$(div).find("#slider").slider("destroy");
334 			$(div).find("#input_value").off();
335 		}
336 	},
337 
338 	/**
339 	 * It returns the value for an element inside a chooser
340 	 * @param {string} chooser The id of the chooser
341 	 * @param {string} element The id of the element
342 	 * @return {number} The value of the element inside that chooser
343 	 */
344 	fetch: function(chooser, element) {
345 		return $("#" + chooser + "_chooser #element_" + element + " #input_value").val();
346 	},
347 
348 	/**
349 	 * It returns the all the values of the elements inside a chooser
350 	 * @param {string} chooser The id of the chooser
351 	 * @return {number} The values of all the chooser's elements
352 	 */
353 	fetchAll: function(chooser) {
354 		var all = $("#" + chooser + "_chooser").find(".element"),
355 			values = {};
356 
357 		for (var i = 0; i <= all.length - 1; i++) {
358 			values[$(all[i]).attr("id")] = $(all[i]).find("#input_value").val();
359 		}
360 		return values;
361 	}
362 };
