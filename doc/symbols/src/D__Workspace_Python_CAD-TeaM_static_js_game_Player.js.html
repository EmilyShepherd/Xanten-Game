  9 /**
 10  * A player contains information regarding the current player: resources, city, name and id
 11  * @constructor
 12  * @param (object) info 		The object which holds information regarding the player (it must have id, name, resources, and buildings).	
 13  * @returns (object) Player		The player
 14  */
 15 function Player(info){
 16 	
 17 	this.id			= info.id; 
 18 	this.name		= info.name;
 19 	this.resources	= new Resources((info.resources?info.resources:{}));	
 20 	this.city 		= (info.buildings)?(new City(info.buildings)):{};
 21 };
 22 
 23 /**
 24  * It returns the level of player. The level of player is the same as the level of city (the level of city is given by the level of administration building)
 25  * @return {number} level The level of the player
 26  * @memberOf Player.prototype
 27  */
 28 Player.prototype.getLevel = function(){
 29 	return this.city.getLevel();
 30 };
 31 
 32 /**
 33  * It consumes the given resources: gold, food, stone and wood, people from administration (free people) and people from military. Then, it updates the window
 34  * @param {Resources} resourcesToConsume An Resources object which holds the resources to be consumed
 35  * @see Resources
 36  * @memberOf Player.prototype
 37  */
 38 Player.prototype.consumeResources = function(resourcesToConsume){
 39 	if(resourcesToConsume.resources) {
 40 		for(resource in resourcesToConsume.resources){
 41 			if(resource === 'military' ) {
 42 				this.city.buildings.military.people -= resourcesToConsume.resources[resource];
 43 			} else {
 44 				this.resources[resource] -= resourcesToConsume.resources[resource];
 45 			}
 46 		}
 47 	}
 48 	if(resourcesToConsume.people) {
 49 		this.city.buildings.administration.people -= resourcesToConsume.people;
 50 	}
 51 	Window.updateResources();
 52 	Window.updateStatistics();
 53 };
 54 
 55 /**
 56  * It adds the given resources: gold, food, stone and wood, people from administration (free people) and people from military. Then, it updates the window
 57  * @param {Resources} resourcesToConsume An Resources object which holds the resources to be added
 58  * @see Resources
 59  * @memberOf Player.prototype
 60  */
 61 Player.prototype.giveResources = function(resourcesToAdd){
 62 	if(resourcesToAdd.resources) {
 63 		for(resource in resourcesToAdd.resources){
 64 			if(resource === 'military' ) {
 65 				this.city.buildings.military.people = parseInt(this.city.buildings.military.people) + parseInt(resourcesToAdd.resources[resource]);
 66 			} else {
 67 				this.resources[resource] = parseInt(this.resources[resource]) + parseInt(resourcesToAdd.resources[resource]);
 68 			}
 69 		}
 70 	}
 71 	if(resourcesToAdd.people) {
 72 		this.city.buildings.administration.people =  parseInt(this.city.buildings.administration.people) + parseInt(resourcesToAdd.people);
 73 	}
 74 	Window.updateResources();
 75 	Window.updateStatistics();
 76 };
