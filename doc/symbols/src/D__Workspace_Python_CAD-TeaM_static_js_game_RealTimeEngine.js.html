  7 /**
  8  * It does the magic of the game. It maintain the connection with the server, it progress any websockets and it progress the resources.
  9  * @constructor
 10  * @property {boolean} isRunning If true, the RTE is running, false otherwise
 11  * @property {object} websocket The reference to the websocket object
 12  */
 13 function RealTimeEngine(token){
 14 	this.isRunning 	= false;
 15 	this.threads 	= {};
 16 	this.websocket  = undefined;
 17 
 18     if (token)
 19 	   this.init(token);
 20 }

 23 /**
 24  * It loads the WebSocket framework and assign the listeners for events (receive/send message, disasters and statistics)
 25  * @private
 26  * @memberOf RealTimeEngine.prototype
 27  */
 28 RealTimeEngine.prototype.init = function(token){
 29     this.channel   = new goog.appengine.Channel(token);
 30     this.socket    = this.channel.open();
 31     this.socket.RTE = this;
 32     this.socket.onopen = this.onopen;
 33     this.socket.onmessage = this.onmessage;
 34     this.socket.onerror = this.onerror;
 35     this.socket.onclose = this.onclose;
 36 };
 37 
 38 RealTimeEngine.prototype.onmessage = function(data)
 39 {
 40     data = JSON.parse(data.data);
 41 
 42     this.RTE[data.call](data.data);
 43 };
 44 
 45 RealTimeEngine.prototype.onopen = function()
 46 {
 47     this.RTE.connected = true;
 48 };
 49 
 50 RealTimeEngine.prototype.onerror = function(error)
 51 {
 52     console.error(error.description);
 53 };
 54 
 55 RealTimeEngine.prototype.onclose = function()
 56 {
 57     this.RTE.connected = false;
 58 };

114 /**
115  * It is called each second. It updates the amounts of resources
116  * @memberOf RealTimeEngine.prototype
117  * @private
118  */
119 RealTimeEngine.prototype._progress = function(){
120 	// food 
121 	game.player.resources.food 	+= parseInt(game.constructions.buildings.farm.capacity(game.player.city.buildings.farm.level).resources.food);
122 	game.player.resources.food 	+= parseInt(game.constructions.buildings.mill.capacity(game.player.city.buildings.mill.level).resources.food);
123 	// wood
124 	game.player.resources.wood 	+= parseInt(game.constructions.buildings.lumberjack.capacity(game.player.city.buildings.lumberjack.level).resources.wood);
125 	// stone
126 	game.player.resources.stone += parseInt(game.constructions.buildings.mine.capacity(game.player.city.buildings.mine.level).resources.stone);
127 	
128 	Window.updateResources();
129 };

184 /**
185  * It is called when the game receives the statistics for a day. It includes disasters
186  * @memberOf RealTimeEngine.prototype
187  * @param {object} statistics The statistics for a day (number of free people, resources.gold)
188  */
189 RealTimeEngine.prototype.receiveDailyStatistics = function(statistics){
190 	Window.newsBoard.add("<span class='news_done'>Done</span>:  Daily statistics: <br />" + HTML_Engine.content.displaysResources(statistics));
191 	game.player.resources.gold 							= statistics.resources.gold;
192 	game.player.city.buildings.administration.people 	= statistics.people;
193 };

195 /**
196  * It is called when server tells the status of the game
197  * @memberOf RealTimeEngine.prototype
198  * @param {string} status It contains the status of the game. If it is lost, it stops the game
199  */
200 RealTimeEngine.prototype.receiveGameStatus = function(status){
201 	if(status === 'lost') {
202 		game.performAction("game_over");
203 		game.freeze();
204 	}
