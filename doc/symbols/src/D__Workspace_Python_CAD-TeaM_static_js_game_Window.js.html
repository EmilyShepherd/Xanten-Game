  6 /**
  7  * It represents the view object which updates information
  8  * @constructor
  9  */
 10 function Window() {}
 11 
 12 /**
 13  * A reference to the bord for actions
 14  * @memberOf Window
 15  */
 16 Window.actionsBoard = new Board("actions");
 17 
 18 /**
 19  * A reference to the board for news
 20  * @memberOf Window
 21  */
 22 Window.newsBoard = new Board("news");
 23 
 24 /**
 25  * A reference to the board for tasks
 26  * @memberOf Window
 27  */
 28 Window.currentTasksBoard = new Board("tasks");
 29 
 30 /**
 31  * It scrools the window to top, it renders the game and sets a listener for changing the map buttons
 32  * @memberOf Window
 33  */
 34 Window.init = function() {
 35 	window.scrollTo(0, 0);
 36 	$(window).resize(Window.render);
 37 	$("#map-view").buttonset();
 38 	$('#map-view :radio').change(function() {
 39 		game.selectMap(game[$('#map-view :radio:checked').val() + "Map"]);
 40 	});
 41 	$("#news_clear").click(function() {
 42 		Window.newsBoard.clear();
 43 	});
 44 	Window.update();
 45 	Window.refresh();
 46 };
 47 
 48 /**
 49  * It updates the details of city, resources, statistics
 50  * @memberOf Window
 51  */
 52 Window.refresh = function() {
 53 	Window.updateDetailsCity();
 54 	Window.updateResources();
 55 	Window.updateStatistics();
 56 };
 57 
 58 /**
 59  * It updates the resources (gold, stone, wood, food)
 60  * @memberOf Window
 61  */
 62 Window.updateResources = function() {
 63 	$("#resources #stone").html(HTML_Engine.shortResourceRepresentation(game.player.resources.stone));
 64 	$("#resources #wood").html(HTML_Engine.shortResourceRepresentation(game.player.resources.wood));
 65 	$("#resources #food").html(HTML_Engine.shortResourceRepresentation(game.player.resources.food));
 66 };
 67 
 68 /**
 69  * It updates the number of people
 70  * @memberOf Window
 71  */
 72 Window.updateStatistics = function() {
 73 	$("#resources #people").html(HTML_Engine.shortResourceRepresentation(game.player.city.getNumberOfPeople()));
 74 	$("#resources #gold").html(HTML_Engine.shortResourceRepresentation(game.player.resources.gold));
 75 };
 76 
 77 /**
 78  * It updates the details of the city (name, level, type of city)
 79  * @memberOf Window
 80  */
 81 Window.updateDetailsCity = function() {
 82 	$("#city_details #name").html(game.player.name);
 83 	$("#city_details #level").html("Level: " + game.player.getLevel());
 84 	$("#city_details #type").html(game.organization.getByLevel("name", game.player.getLevel()));
 85 };
 86 
 87 /**
 88  * It updates the width and height for all boards, windows
 89  * @memberOf Window
 90  */
 91 Window.update = function() {
 92 	$("#actions_board").css({
 93 		"height": 1 + "px",
 94 		"min-height": 1 + "px"
 95 	});
 96 	$("#news_board").css({
 97 		"height": 1 + "px",
 98 		"min-height": 1 + "px"
 99 	});
100 	$("#tasks_board").css({
101 		"height": 1 + "px",
102 		"min-height": 1 + "px"
103 	});
104 
105 	var height = $(document).height();
106 
107 	var heightForOneBoard = height - 245;
108 	var heightForTwoBoards = (height - (105 + 135 * 2)) / 2;
109 
110 	$("#actions_board").css({
111 		"height": heightForOneBoard + "px",
112 		"min-height": heightForOneBoard + "px"
113 	});
114 	$("#news_board").css({
115 		"height": heightForTwoBoards + "px",
116 		"min-height": heightForTwoBoards + "px"
117 	});
118 	$("#tasks_board").css({
119 		"height": heightForTwoBoards + "px",
120 		"min-height": heightForTwoBoards + "px"
121 	});
122 };
