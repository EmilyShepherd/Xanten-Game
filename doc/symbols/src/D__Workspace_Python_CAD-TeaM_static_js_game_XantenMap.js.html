  7 /**
  8  * The map object
  9  * @constructor
 10  * @param {array} mapArray A 2d array which contains information regarding the map
 11  * @param {string} name It should be city or world
 12  */
 13 function XantenMap(array, name){
 14 	this.array 				= array;
 15 	this.selectedCell 		= null;
 16 	this.name 				= name;
 17 	this.HTML_element		= "#map-board-"+this.name;

 19 }

103 /**
104  * It creates the table for the map and it links the listeners for the squares. It is called ones
105  * @memberOf XantenMap.prototype
106  */
107 XantenMap.prototype.init = function(){	
108 	
109 	var instance = this;	// save a reference for lambda functions
110 	var table    = '';
111 	var div 	 = '';
112 	
113 	table = "<div id='map-board-" + this.name + "' class='map' >";
114 	
115 	for(var vertical = 1; vertical <= this.getSize(); vertical++){
116 		table += '<div class="row">';
117 		for(var horizontal = 1; horizontal <= this.getSize(); horizontal++){
118 			table += '<div id="cel_' + vertical + '_' + horizontal + '"></div>';
119 		}
120 		table += '</div>';
121 	}
122 	
123 	table 	+= "</div>";
124 	div 	= "<div id='map-window-" + this.name + "' class='map-window'>" + table + "</div>";
125 	$("#map-window").append(div);
126 	$("#map-board-"+this.name+" .row > div").click(function(){
127 		var id = $(this).attr("id");
128 		var array = id.substring(4, id.length);
129 		var elements = array.split("_");
130 		instance._selectCell(elements[0], elements[1]);
131 	});	
132 	if(this._init){
133 		this._init();
134 	}	
135 	this.render();
136 };

139 /**
140  * It is called when the map is selected. It calls the render and shows the map
141  * @memberOf XantenMap.prototype
142  */
143 XantenMap.prototype.select = function(){	
144 	$(this.HTML_element).fadeIn("slow");
145 	$("#map-window-"+this.name).show();
146 	$("#map-board-city-"+this.name).show();
147 	if(this._select){
148 		this._select();
149 	}
150 };

152 /**
153  * It is called when the map is hidden. It calls the deselect method and then hides the map
154  * @memberOf XantenMap.prototype
155  */
156 XantenMap.prototype.hide = function(){
157 	this.deselect();
158 	$(this.HTML_element).hide();
159 	if(this._hide){
160 		this._hide();
161 	}
162 };

164 /**
165  * It saves the selected cell and calls selectCell method
166  * @memberOf XantenMap.prototype
167  * @param {number} x The x coordinate
168  * @param {number} y The y coordinate
169  */
170 XantenMap.prototype._selectCell = function(x, y){
171 	
172 	this.deselect();
173 	this.selectCell(x, y);
174 	this.selectedCell = {"x": x, "y": y};
175 };

177 /**
178  * It changes the map backgrounds and images.
179  * @memberOf XantenMap.prototype
180  */
181 XantenMap.prototype.update = function(){	
182 	var s = this.selectedCell;
183 	this.render();
184 	if(s) {
185 		this._selectCell(parseInt(s.x), parseInt(s.y));
186 	}
187 };
