445 /**
446  * It returns the data object of a building using the id of the building
447  * @param {number} id The id of the building
448  * @return {object} The object which contain information regarding the building
449  * @memberOf Game.prototype
450  */
451 game.constructions.getBuildingById = function(id) {
452 	for (b in game.constructions.buildings) {
453 		if (game.constructions.buildings[b].id === id) {
454 			return game.constructions.buildings[b];
455 		}
456 	}
457 };
