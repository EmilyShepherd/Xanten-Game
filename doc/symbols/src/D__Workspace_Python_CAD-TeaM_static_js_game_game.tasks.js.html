 13 	// move units to or from a enemy city
 14 	"move_military_units": function(data) {
 15 		return new Task(data,
 16 			'Move <span class="bold">' + data.options.element_units + '</span> military units<br />' + "<span class='bold'>" + HTML_Engine.worldPath.getCityName(data.from) + "</span> --> <span class='bold'>" + HTML_Engine.worldPath.getCityName(data.to) + "</span>", {
 17 				"url": '/military/attack/' + data.to + '/' + data.options.element_units,
 18 				"type": 'GET'
 19 			},
 20 			function(task) {
 21 				if (task.data.to !== game.player.id) {
 22 					// Attack - Stage 1
 23 					//
 24 					// Your_city ---> Other
 25 					var resources = game.unit.military.attack(task.data.options.element_units);
 26 					game.player.city.buildings.military.people -= task.data.options.element_units;
 27 					game.player.consumeResources(resources);
 28 					game.update();
 29 
 30 				} else {
 31 					// Attack - Stage 3
 32 					//
 33 					// Other ---> Your_city 
 34 					task.response = {
 35 						friends: {
 36 							resources: {
 37 								"military": 45,
 38 								"food": 20001
 39 							}
 40 						},
 41 						enemies: {
 42 							resources: {
 43 								"military": 40,
 44 								"food": 200
 45 							}
 46 						},
 47 						status: "won",
 48 						carring: {
 49 							resources: {
 50 								"food": 200,
 51 								"wood": 4000,
 52 								"gold": 2000,
 53 								"military": 10
 54 							}
 55 						}
 56 					};
 57 
 58 					if (task.response.status === 'won' && task.response.carring.resources.military) {
 59 						task.data.options.element_units = task.response.carring.resources.military;
 60 						task.title = 'Returning <span class="bold">' + data.options.element_units + '</span> military units<br />' + "<span class='bold'>" + HTML_Engine.worldPath.getCityName(data.from) + "</span> --> <span class='bold'>" + HTML_Engine.worldPath.getCityName(data.to) + "</span>";
 61 						// report
 62 						Window.newsBoard.add(
 63 							HTML_Engine.attackCity.report(task.data, task.response)
 64 						);
 65 					} else {
 66 						// report
 67 						Window.newsBoard.add(
 68 							HTML_Engine.attackCity.report(task.data, task.response)
 69 						);
 70 						task.forceStop();
 71 					}
 72 				}
 73 			},
 74 			function(task) {
 75 				if (task.data.to === game.player.id) {
 76 					// Attack - Stage 2
 77 					//					
 78 					// It reached your city. Finished the attack
 79 
 80 					task.response.carring = new Resources(task.response.carring);
 81 					game.player.giveResources(task.response.carring);
 82 
 83 
 84 
 85 				} else {
 86 					// Attack - Stage 4
 87 					//
 88 					// It reached the destination. Now it comes back to city
 89 					var data = task.data,
 90 						temp = task.data.from;
 91 
 92 					data.response = task.response;
 93 					data.from = task.data.to;
 94 					data.to = temp;
 95 
 96 					game.performTask("move_military_units", data);
 97 
 98 				}
 99 			},
100 			"static/img/game/resource/military.png");
101 	},

103 	"trade": function(data) {
104 		return new Task(data,
105 			'Trade ' + data.what + ' for <span class="bold">' + data["for"] + "</span>", {
106 				// TODO change url 
107 				"url": '/me/building/' + data.building + '/build',
108 				"data": data,
109 				"type": 'GET'
110 			},
111 			function(task) {
112 				task.data.receive = new Resources({
113 					resources: {}
114 				});
115 				task.data.give = new Resources({
116 					resources: {}
117 				});
118 				if (data['for'] === 'gold') {
119 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_wood, 0.10, "gold"));
120 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_food, 0.50, "gold"));
121 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_stone, 0.20, "gold"));
122 					task.data.give.people = task.data.receive.people;
123 					task.data.give.resources.wood = task.data.resources.element_wood;
124 					task.data.give.resources.food = task.data.resources.element_food;
125 					task.data.give.resources.stone = task.data.resources.element_stone;
126 				} else {
127 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_wood, 5, "wood"));
128 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_food, 25, "food"));
129 					task.data.receive = Resources.combine(task.data.receive, game.unit.trade(task.data.resources.element_stone, 10, "stone"));
130 					task.data.give.resources.gold = parseInt(task.data.resources.element_wood) + parseInt(task.data.resources.element_stone) + parseInt(task.data.resources.element_food);
131 					task.data.give.people = task.data.receive.people;
132 				}
133 				game.player.consumeResources(task.data.give);
134 			},
135 			function(task) {
136 				game.player.giveResources(task.data.receive);
137 			},
138 			"static/img/game/buildings/trade.png");
139 	},

335 	"create_building": function(args) {
336 		var data = {};
337 		data.cell = game.cityMap.getSelectedCell();
338 		data.building = $(args).attr("building_name");
339 
340 		return new Task(data,
341 			'Create building ' + data.building, {
342 				"url": '/me/building/' + data.building + '/build',
343 				"type": 'GET'
344 			},
345 			function(task) {
346 				// create building
347 				if (task.data.building !== "house") {
348 					game.player.city.buildings[task.data.building].level = 1;
349 				}
350 				var resources = game.constructions.buildings[task.data.building].levelUp(1);
351 				task.data.peopleBack = resources.people;
352 				game.player.city.buildings[task.data.building].status = 'under_construction';
353 				game.cityMap.array[data.cell.x - 1][data.cell.y - 1].type_construction = "building";
354 				game.cityMap.array[data.cell.x - 1][data.cell.y - 1].id_construction = game.constructions.buildings[task.data.building].id;
355 				game.cityMap.update(); // change the status of the city map
356 				game.player.consumeResources(resources);
357 			},
358 			function(task) {
359 				game.player.city.buildings[task.data.building].num++;
360 				game.player.city.buildings[task.data.building].status = 'Done';
361 				game.player.city.buildings.administration.people = parseInt(game.player.city.buildings.administration.people) + task.data.peopleBack;
362 			},
363 			"/static/img/game/buildings/" + data.building + ".png");
364 	},
