  8 /**
  9  *  An action object is a model (MCV) object which performs an action of the user.
 10  *  @constructor
 11  *  @param {HTML_Engine} HTML_Engine_Reference The function which generates the content (usually a HTML_Engine object) 
 12  *  @param {function} beforeRemove A function to be called before the Action is done (or removed by the game)
 13  *  @param {function} serverCall If the action needs to get remote information and to execute a function after receiving the inforamation, this value should be an object for the AJAX request. The object MUST contain "type", "url", "data". You can specify a callback function using the parameter "cb" 
 14  * 	@example 
 15  * 		new Action("Actions city", HTML_Engine.selectCity, function() {
 16  *			game.worldMap.deselect();
 17  *		});
 18  */
 19 function Action(name, HTML_Engine_Reference, beforeRemove, serverCall){	
 20 	this.name				   = name;
 21 	this.instance			   = this;
 22 	this.HTML_Engine_Reference = HTML_Engine_Reference;
 23 	this.beforeRemove          = beforeRemove;
 24 	this.args                  = undefined;
 25 	this.serverCall            = serverCall?serverCall:undefined;
 26 	this.ajaxCall			   = undefined;
 27 }
 28 
 29 /**
 30  * It sets the arguments for the action
 31  * @param {object} args The arguments for the action
 32  */
 33 Action.prototype.setArguments = function(args){
 34 	this.args = args;
 35 };
 36 
 37 /**
 38  * It calls the perform method of the action and refreshes the content of the action. It does fadeOUT - fadeIn
 39  * @memberOf Action.prototype
 40  */
 41 Action.prototype.update = function(){
 42 	this.perform(false);
 43 }
 44 
 45 
 46 /**
 47  * If the action needs an AJAX request it does it. Then it calls the call back and updates the HTML. If it does not need an AJAX request, it just updates the HTML
 48  * @param {boolean} fade If set to true, the action will be shown by a fade effect
 49  * @memberOf Action.prototype
 50  */
 51 Action.prototype.perform = function(fade){
 52 	var instance = this;
 53 	$("#action_window div.before_window").html(this.name);
 54 	var instance   = this;
 55 	if(this.serverCall){
 56 		HTML_Engine.loadAction.content()
 57 		this.ajaxCall = $.ajax({"url": this.serverCall.url,
 58 				"type": this.serverCall.type,
 59 				"data": this.serverCall.data,
 60 				"success": function(response){instance.serverCall.cb(response);instance.updateHTML(true);},
 61 				"error": function(){HTML_Engine.failAction.content(); HTML_Engine.failAction.enable();}
 62 		});
 63 	} else {
 64 		this.updateHTML(fade);	
 65 	}
 66 };
 67 
 68 /**
 69  * It updates the HTML for the action by calling the content function of the HTML_Engine object. Then it activates the content.
 70  * @param {boolean} fade If set to true, the action will be shown by a fade effect
 71  * @private 
 72  * @memberOf Action.prototype
 73  */
 74 Action.prototype.updateHTML = function(fade){
 75 	if(fade) {
 76 		$("#actions_board .inside").html("");
 77 		$("#actions_board .inside").hide();
 78 	}
 79 	$("#actions_board .inside").html(this.HTML_Engine_Reference.content?this.HTML_Engine_Reference.content(this.args):"");
 80 	if(fade) {
 81 		$("#actions_board .inside").fadeIn('slow');
 82 	}
 83 	
 84 	if(this.HTML_Engine_Reference.enable){
 85 		this.HTML_Engine_Reference.enable(this.args);
 86 	}
 87 };
 88 
 89 /**
 90  * If there was any AJAX request it aborts it. If the action has code to be executed before the function is done, it executes it. Then, it disables the HTML listeners by calling the 'disable' function of the HTML_Engine object
 91  * @memberOf Action.prototype
 92  */
 93 Action.prototype.remove = function(){
 94 	if(this.ajaxCall){
 95 		this.ajaxCall.abort();
 96 	}
 97 	
 98 	if(this.beforeRemove){
 99 		this.beforeRemove();
100 	}
101 	
102 	if(this.HTML_Engine_Reference.disable){
103 		this.HTML_Engine_Reference.disable();
104 	}
105 };
