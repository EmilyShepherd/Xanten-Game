  9 /**
 10  * The bord object represents a HTML div where there is added content
 11  * @constructor
 12  * @param {string} id The id of the board. It can be 'actions', 'news' or 'tasks'
 13  */
 14 function Board(id){
 15 	this.id = id;
 16 }
 17 
 18 /**
 19  * It adds a message to the board
 20  * @memberOf Board.prototype
 21  * @param {string} message The string to be added
 22  */
 23 Board.prototype.add = function(message){
 24 	var div = $("<div style='display:none'>"+message+"</div>");
 25 	$("#"+this.id+"_board").prepend(div);
 26 	div.fadeIn(500);;
 27 }
 28 
 29 /**
 30  * It clears the board
 31  * @memberOf Board.prototype
 32  */
 33 Board.prototype.clear = function(){
 34 	var divs = $("#"+this.id+"_board").find("div");
 35 	for(var i=0; i<divs.length; i++){
 36 		var div = divs[i];
 37 		$(div).fadeOut(i*100, function() { $(this).remove(); });	
 38 	}
 39 }
