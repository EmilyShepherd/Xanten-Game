 18 /**
 19  * It sums the number of all people for the city
 20  * @returns {number} The number of all the people from the city
 21  * @memberOf City.prototype
 22  */
 23 City.prototype.getNumberOfPeople = function() {
 24 	var people = 0;
 25 	for(building in this.buildings){
 26 		if(this.buildings[building].people) {
 27 			people += parseInt(this.buildings[building].people);
 28 		}
 29 	}	
 30 	return people;
 31 };
