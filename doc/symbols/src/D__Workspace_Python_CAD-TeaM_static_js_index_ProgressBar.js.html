  7 /**
  8  * Represents performing task. It is a view object (MCV). It shows the progress of task in real time in the current task box
  9  * @construct
 10  * @property {Task} task The reference of the task object
 11  * @property {object} p The progress object
 12  * @param Task task The task object
 13  */
 14 function ProgressBar(task){
 15 	
 16 	this.task = task;
 17 	
 18 	Window.currentTasksBoard.add("<div id='task_"+this.task.id+"'><img width='25px' height='25px' src='"+task.imgSource+"' align='absmiddle' >   "+task.title+"<div id='progressbar_"+this.task.id+"'><div class='progress-label'></div></div></div>");
 19 	
 20 	this.p			 		= $( "#progressbar_"+this.task.id );
 21     this.progressLabel 		= $( "#progressbar_"+this.task.id ).find( ".progress-label" );
 22     
 23     var instance			= this,
 24 	    seconds 			= Math.random() * (25 - 7) + 7;
 25 	 	step 				= seconds * 10,
 26 	 	progress			= (function(){ var i = instance; 
 27 								 	return function(){
 28 							    		val = i.p.progressbar( "value" ) || 0;
 29 							    		i.p.progressbar( "value", val + 1 );	
 30 							    	}; })(instance);
 31  
 32     this.p.progressbar({
 33 		value: false,
 34 		change: function() {
 35     		instance.progressLabel.text( instance.p.progressbar( "value" ) + "%" );
 36 		},
 37 		complete: function() {
 38 			instance.progressLabel.text( "Complete!" );
 39 			instance.finish();
 40 		}
 41     });	
 42 	
 43 }
 44 
 45 /**
 46  *  It finished the task. It stops the progress, calls the task.afterEnds() method and deletes itself from the game
 47  *  @memberOf ProgressBar.prototype
 48  */
 49 ProgressBar.prototype.finish = function() {	
 50 	this.task.afterEnds();
 51 };
 52 
 53 /**
 54  * It starts the thread
 55  * @memberOf ProgressBar.prototype
 56  */
 57 ProgressBar.prototype.start = function(){
 58 	this.thread = setInterval(progress, step);
 59 }
