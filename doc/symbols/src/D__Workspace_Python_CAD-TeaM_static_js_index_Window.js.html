 31 /**
 32  * It scrools the window to top, it renders the game and sets a listener for changing the map buttons
 33  * @memberOf Window
 34  */
 35 Window.init = function() {	
 36 	window.scrollTo(0, 0);
 37 	console.log('here')
 38 	$(window).resize(Window.render);	
 39 	$("#map-view").buttonset();
 40 	$('#map-view :radio').change(function() {
 41 		game.selectMap(game[$('#map-view :radio:checked').val() + "Map"]);
 42 	});
 43 	Window.updateDetailsCity();
 44 	Window.updateResources();
 45 	Window.updateStatistics();
 46 	Window.update();
 47 }
 48 
 49 /**
 50  * It updates the resources (gold, stone, wood, food)
 51  * @memberOf Window
 52  */
 53 Window.updateResources = function(){
 54 	$("#resources #stone").html(HTML_Engine.shortResourceRepresentation(game.player.resources.stone));
 55 	$("#resources #wood").html(HTML_Engine.shortResourceRepresentation(game.player.resources.wood));
 56 	$("#resources #food").html(HTML_Engine.shortResourceRepresentation(game.player.resources.food));
 57 };
 58 
 59 /**
 60  * It updates the number of people
 61  * @memberOf Window
 62  */
 63 Window.updateStatistics = function() {
 64 	$("#resources #people").html(HTML_Engine.shortResourceRepresentation(game.player.city.getNumberOfPeople()));
 65 	$("#resources #gold").html(HTML_Engine.shortResourceRepresentation(game.player.resources.gold));
 66 };
 67 
 68 /**
 69  * It updates the details of the city (name, level, type of city)
 70  * @memberOf Window
 71  */
 72 Window.updateDetailsCity = function() {
 73 	$("#city_details #name").html(game.player.name);
 74 	$("#city_details #level").html("Level: "+game.player.level);
 75 	$("#city_details #type").html(game.organization.getByLevel("name", game.player.level) );
 76 };

 78 /**
 79  * It updates the width and height for all boards, windows
 80  * @memberOf Window
 81  */
 82 Window.update = function() {
 83 	$("#actions_board").css({"height":1+"px", "min-height": 1+"px"});
 84 	$("#news_board").css({"height":1+"px", "min-height": 1+"px"});
 85 	$("#tasks_board").css({"height":1+"px", "min-height": 1+"px"});
 86 	
 87 	var height = $(document).height();
 88 	
 89 	var heightForOneBoard = height-245;
 90 	var heightForTwoBoards = (height-(105+135*2))/2;
 91 	
 92 	$("#actions_board").css({"height":heightForOneBoard+"px", "min-height": heightForOneBoard+"px"});
 93 	$("#news_board").css({"height":heightForTwoBoards+"px", "min-height": heightForTwoBoards+"px"});
 94 	$("#tasks_board").css({"height":heightForTwoBoards+"px", "min-height": heightForTwoBoards+"px"});
 95 }
