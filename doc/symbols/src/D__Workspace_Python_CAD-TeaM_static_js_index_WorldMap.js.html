109 	/**
110 	 * It adds a new path
111 	 * @param {array} path A 2d array with the path
112 	 * @param {string} type It can be 'attack' or 'trade'
113 	 * @param {string} description The description for the path
114 	 * @param {number} duration The duration in ms
115 	 * @param {function} callBack A function to be called when the path is removed
116 	 * @private
117 	 */
118 	WorldMap._addPath = function (path, type, description, duration, callBack){
119 			
120 		if(!this.canvas.paths.length){
121 			this._drawCanvas();
122 		}
123 		
124 		var id 							= Object.keys(this.canvas.paths).length
125 			instance 					= this, //make the variables global, so you can access them in the animation function
126 			ido 						= "path_"+(parseInt(id) + 0),
127 			instance.canvas.paths[ido]	= {},
128 			currentPath					= this.canvas.paths[ido],
129 			currentPath.duration		= duration,
130 			currentPath.id				= ido,
131 			pixel						= 100,
132 			startX						= path[0][0] * pixel + 50,
133 			startY						= path[0][1] * pixel + 50,
134 			lines						= [startX, startY],
135 			x							= startX,
136 			y							= startY;;
137 			
138 		if(path.length%2 === 0){
139 			lines.push(startX);
140 			lines.push(startY);
141 		}
142 			
143 		var circle = new createjs.Shape();
144 			circle.graphics.beginFill("DeepSkyBlue").drawCircle(0, 0, 2);
145 			circle.x = startX;
146 			circle.y = startY;
147 		
148 		var line = new createjs.Shape();
149 			line.graphics.setStrokeStyle(3);
150 			line.graphics.beginStroke("#A7ACA6");
151 			line.graphics.moveTo(startX, startY);
152 			
153 		for (var i = 1; i <= path.length - 1; i++) {
154 			
155 
156 			var zx = path[i][0];
157 			var zy = path[i][1];
158 
159 			var small_line = [zx * pixel + 50, zy * pixel + 50];
160 			
161 			lines.push(small_line[0]);
162 			lines.push(small_line[1]);
163 			line.graphics.lineTo(small_line[0], small_line[1]);
164 			
165 
166 			x = path[i][0];
167 			y = path[i][1];
168 		}
169 		
170 				
171 		line.graphics.endStroke();
172 		
173 		var end = (function(){
174 			var v = ido;
175 			if(callBack){
176 				var c = callBack;
177 			}
178 			return function(){instance.removePath(v); if(c) {c();} };
179 		})();
180 		
181 		createjs.Tween.
182 				get(circle).
183 				to({guide:{ path:lines}},duration).
184 				call(end);
185 		
186 	    
187 		this.canvas.stage.addChild(line);
188 		this.canvas.stage.addChild(circle);
189 
190 		
191 		this.canvas.stage.update();
192 		
193 		currentPath.line 	= line;
194 		currentPath.circle 	= circle;     
195 	};

197 	/**
198 	 * It removes the path
199 	 * @param {number} id The id of the path (eg. path_0)
200 	 * @private 
201 	 */
202 	WorldMap.removePath = function(id){
203 		var p =game.worldMap.canvas.paths[id];
204 		p.circle.removeAllEventListeners();
205 		this.canvas.stage.removeChild(p.line)		
206 		this.canvas.stage.removeChild(p.circle)		
207 		this.canvas.stage.update();
208 		delete this.canvas.paths[id];		
209 		if(!this.canvas.paths.length){
210 			createjs.Ticker.removeAllEventListeners();
211 		}
212 	};

268 	/**
269 	 * It deletes all the paths and stops the ticker listener
270 	 * @memberOf WorldMap.prototype
271 	 */
272 	WorldMap._freeze = function(){
273 		for(path in this.canvas.paths){
274 			this.removePath(this.canvas.paths[path].id);
275 		}
276 		createjs.Ticker.removeAllEventListeners();
277 	}
