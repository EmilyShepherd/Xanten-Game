  9 /**
 10  * It represents a WorldMap object. It extends the XantenMap object
 11  * @constructor
 12  * @extends XantenMap
 13  * @param array The array of the map
 14 */
 15 function WorldMap(obj){
 16 	
 17 	var WorldMap	 		= new XantenMap(obj.array, 'world');
 18 	WorldMap.__proto__ 		= "WorldMap";
 19 	WorldMap.players		= obj.players;
 20 	WorldMap.backgrounds	= obj.backgrounds;
 21 	
 22 	/**
 23 	 * 
 24 	 * @memberOf WorldMap.prototype 
 25 	 * @param x The x coordinate
 26 	 * @param y The y coordinate
 27 	 */
 28 	WorldMap.selectCell = function(x, y){
 29 		
 30 		var id_selected_city = this.getCityByPosition(x, y);
 31 		if(id_selected_city){
 32 			if(id_selected_city !== game.player.id){
 33 				game.performAction("selectCity", id_selected_city);
 34 			} else {
 35 				game.performAction("city-map-selected");
 36 			}
 37 			this.selectImage(x, y);
 38 		} else {
 39 			game.performAction('clear', null);
 40 		}
 41 		
 42 		// alert('You hit the square '+x+' '+y+ ' from world map. This map has no selectCell method yet')
 43 		// TODO waiting for python team
 44 	};
 45 
 46 	/**
 47 	 * It renders the information of the array into Graphical form
 48 	 * @memberOf WorldMap.prototype 
 49 	 */
 50 	WorldMap.render = function(){
 51 		for(i=1; i <= this.array.length; i++){
 52 			for(j=1; j <= this.array.length; j++){
 53 				var cell 		= $(this.HTML_element + " #cel_"+i+"_"+j),
 54 					array_bg	= this.array[(i-1)][(j-1)],
 55 					background 	= this.backgrounds[array_bg.id_background];
 56 				
 57 				cell.removeClass();
 58 				
 59 				if(background.allowCity && array_bg.id_city) {	
 60 					cell.addClass("allow_city");	
 61 					var player 	= this.getCityById(array_bg.id_city);
 62 					cell.html(HTML_Engine.getBuilding.image("administration", player.level, 98) + "<span class='level'>" + player.level + "</span>" + "<span class='description'>"+player.name.capitalize()+"</span>");
 63 					$(this.HTML_element+" #cel_"+i+"_"+j+" img").addClass('hasAction');	
 64 					$(this.HTML_element+" #cel_"+i+"_"+j+" img").attr({"name_of_building": player});	
 65 				} 			
 66 				$(this.HTML_element + " #cel_"+i+"_"+j).css('background-image', 'url(/static/img/game/background/world/' + background.img + ')');
 67 			}
 68 		}		
 69 		// this.renderTrades();
 70 		// this.renderAttacks();
 71 	};
 72 		
 73 	 
 74  	/** 
 75  	 * It returns the city using the id
 76  	 * @param id The id of the city
 77 	 * @return (object) An object with information regarding the city
 78 	 * @memberOf WorldMap.prototype
 79 	 */
 80 	WorldMap.getCityById = function(id){
 81 		return this.players[id];
 82 	};
 83 	
 84 	/**
 85 	 * It returns the id of the city based on the position on the world map
 86 	 * @param (number) x The x position
 87 	 * @param (number) y The y position
 88 	 * @return (number) The id of the city or null if nothing is there
 89 	 * @memberOf WorldMap.prototype
 90 	 */
 91 	WorldMap.getCityByPosition = function(x, y){
 92 		return this.array[(x-1)][(y-1)].id_city;
 93 	};
 94 	
 95 	/**
 96 	 * It returns the position of a city on the map
 97 	 * @param (number) id The id of the city
 98 	 * @return (object) An object with the x and y position
 99 	 * @memberOf WorldMap.prototype
100 	 */
101 	WorldMap.getCityPositionById = function(id){
102 		for(i=0; i<=this.array.length-1; i++){
103 			for(j=0; j<=this.array.length-1; j++){
104 				if(array[i][j].id_city === id){
105 					return {
106 						"x" : i,
107 						"y" : j
108 					}
109 				}
110 			}
111 		}
112 	};	
113

115 	 * It deletes all the paths and stops the ticker listener
116 	 * @memberOf WorldMap.prototype
117 	 */
118 	WorldMap._freeze = function(){
119 	}
120 	
121 	return WorldMap;
122  }
