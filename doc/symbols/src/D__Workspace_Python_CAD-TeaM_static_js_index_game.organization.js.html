1 /**
  2  * Holds information regarding the organization
  3  */
  4 game.organization = {};
  5 
  6 /**
  7  * It returns a particular info regarding the level of an organization
  8  * @param (string) what It can be "name", "administration", "house" or "period"
  9  * @param (string) level The level of organization
 10  * @return (string) The particular information
 11  */
 12 game.organization.getByLevel = function(what, level) {
 13 
 14 	var info = {},
 15 		level = parseInt(level);
 16 
 17 	if (level <= 4) {
 18 		info = {
 19 			name: "Hamlet",
 20 			administration: "Foyer",
 21 			house: "Neighbourhood",
 22 			period: "Ancient Time"
 23 		};
 24 	} else
 25 	if (level >= 5 && level <= 9) {
 26 		info = {
 27 			name: "Village",
 28 			administration: "Village Hall",
 29 			house: "Cul-de-sac",
 30 			period: "Medieval Age"
 31 		};
 32 	} else
 33 	if (level >= 10 && level <= 14) {
 34 		info = {
 35 			name: "Town",
 36 			administration: "Town Square",
 37 			house: "Residential Area",
 38 			period: "Classical Era"
 39 
 40 		};
 41 	} else
 42 	if (level >= 15 && level <= 19) {
 43 		info = {
 44 			name: "City",
 45 			administration: "City Hall",
 46 			house: "Borough",
 47 			period: "Industrial Period"
 48 		};
 49 	} else
 50 	if (level >= 20 && level <= 49) {
 51 		info = {
 52 			name: "Metropolis",
 53 			administration: "Metropolis Hall",
 54 			house: "District",
 55 			period: "Modern History"
 56 		};
 57 	} else {
 58 		info = {
 59 			name: "Magapolis",
 60 			administration: "Government",
 61 			house: "Region",
 62 			period: "Contemporary Period"
 63 		}
 64 	}
 65 
 66 	return info[what];
 67 };
